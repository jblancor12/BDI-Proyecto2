USE `pe`;
DROP function IF EXISTS `get_book_title`;

DELIMITER $$
USE `pe`$$
CREATE FUNCTION `get_book_title`(pid_book INT) 
RETURNS VARCHAR(45)
DETERMINISTIC
BEGIN
DECLARE vcName VARCHAR(45);
SET vcName = (
		SELECT title
        FROM book
        WHERE id_book = pid_book
        );
RETURN vcName;
END$$

DELIMITER ;
